Bestimmen Sie die Bildschirmausgabe des nachstehenden Algorithmus, ohne das Programm jedoch auszuführen:

#include <stdio.h>

void calculate (int* a, int b)
{
    *a = 4;
    b = b + *a;
    printf("%d - %d\n", *a, b);
}

void main()
{
    int a = 3;
    int b = 5;

    calculate(&a, b);
    printf("%d - %d\n", a, b);

    calculate(&b, a);
    printf("%d - %d\n", a, b);
}
